
## 1. lower()
Converts all characters to lowercase.
"HELLO".lower()   # 'hello'

## 2. upper()
Converts all characters to uppercase.
"hello".upper()   # 'HELLO'

## 3. title()
Capitalizes the first letter of each word.

"hello world".title()   # 'Hello World'

## 4. strip()
Removes whitespace (or other characters) from both ends.
"  hello  ".strip()     # 'hello'

## 5. replace(old, new)
Replaces parts of a string.
"good night".replace("night", "morning")  # 'good morning'

## 6. find(substring)
Returns the index of the first occurrence of a substring (or -1 if not found).
"hello".find("e")   # 1

## 7. count(substring)
Counts how many times a substring appears.
"banana".count("a")   # 3

## 8. split(separator)
Splits a string into a list.
"apple,banana,mango".split(",")  # ['apple', 'banana', 'mango']

## 9. join(list)
Joins list elements into a string with a separator.
", ".join(["apple", "banana"])   # 'apple, banana'

## 10. startswith() / endswith()
Checks how a string starts or ends.

"hello".startswith("he")   # True
"world".endswith("ld")     # True
